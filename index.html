/** * Partner Enablement Hub - progress tracker + manager dashboard feed * Bind this to the "Partner onboarding Claude" Google Sheet and deploy as a Web App. * * SETUP / UPDATE: * 1. Open the Sheet, then Extensions > Apps Script. * 2. Replace ALL existing code with this file. Click Save. * 3. IMPORTANT: set DASHBOARD_PASSWORD below to your own secret. * 4. Redeploy so the change goes live: * Deploy > Manage deployments > (your deployment) > edit (pencil) * > Version: New version > Deploy. The /exec URL stays the same. * * The hub POSTs consultant progress here (writes/updates a row). * The manager dashboard GETs the data back, but ONLY when the correct * password is supplied. The password lives here on the server, never in * the public hub page. */ var DASHBOARD_PASSWORD = 'owly'; // <-- set your own secret var SHEET_NAME = 'Progress'; var HEADERS = [ 'Last updated', 'Name', 'Company (partner)', 'Overall %', 'Overall state', 'HSE', 'Advanced Inbox', 'Listening', 'Advanced Analytics', 'Amplify', 'Skill gaps', 'Raw JSON' ]; /* ---- consultant submits progress ---- */ function doPost(e) { try { var data = JSON.parse(e.postData.contents); var sh = getSheet_(); var row = buildRow_(data); var key = keyFor_(data.name, data.company); var values = sh.getDataRange().getValues(); var found = -1; for (var i = 1; i < values.length; i++) { if (keyFor_(values[i][1], values[i][2]) === key) { found = i + 1; break; } } if (found > 0) { sh.getRange(found, 1, 1, row.length).setValues([row]); } else { sh.appendRow(row); } return reply_({ ok: true }, null); } catch (err) { return reply_({ ok: false, error: String(err) }, null); } } /* ---- manager dashboard reads data (password required) ---- */ function doGet(e) { var p = (e && e.parameter) ? e.parameter : {}; if (p.action === 'list') { if (String(p.pw || '') !== DASHBOARD_PASSWORD) { return reply_({ ok: false, error: 'unauthorized' }, p.callback); } return reply_({ ok: true, rows: readRows_() }, p.callback); } return reply_({ ok: true, msg: 'Partner Enablement Hub tracker is live.' }, p.callback); } function readRows_() { var sh = getSheet_(); var v = sh.getDataRange().getValues(); var out = []; for (var i = 1; i < v.length; i++) { out.push({ updated: v[i][0], name: v[i][1], company: v[i][2], overallPct: v[i][3], overallState: v[i][4], hse: v[i][5], inbox: v[i][6], listen: v[i][7], aa: v[i][8], amp: v[i][9], gaps: v[i][10] }); } return out; } function getSheet_() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sh = ss.getSheetByName(SHEET_NAME); if (!sh) { sh = ss.insertSheet(SHEET_NAME); } if (sh.getLastRow() === 0) { sh.appendRow(HEADERS); sh.setFrozenRows(1); sh.getRange(1, 1, 1, HEADERS.length).setFontWeight('bold'); } return sh; } function keyFor_(name, company) { return String(name || '').trim().toLowerCase() + '||' + String(company || '').trim().toLowerCase(); } function productCell_(p) { if (!p) { return 'Not started'; } if (p.ready) { return 'Ready to deliver'; } if (p.started) { return p.pct + '% in progress'; } return 'Not started'; } function buildRow_(d) { var p = d.products || {}; return [ new Date(), d.name || '', d.company || '', (d.overallPct != null ? d.overallPct + '%' : ''), d.overallState || '', productCell_(p.hse), productCell_(p.inbox), productCell_(p.listen), productCell_(p.aa), productCell_(p.amp), (d.gaps || []).join(', '), JSON.stringify(d) ]; } /* JSON, or JSONP when a callback is given (used by the dashboard) */ function reply_(obj, callback) { var s = JSON.stringify(obj); if (callback) { var safe = String(callback).replace(/[^A-Za-z0-9_]/g, ''); if (safe) { return ContentService.createTextOutput(safe + '(' + s + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } } return ContentService.createTextOutput(s).setMimeType(ContentService.MimeType.JSON); }
